* Installation
** Nix
This repo is a nix flake.  You can just add ~overlays.default~ output to your
nixpkgs overlays.

#+begin_src nix
{
  inputs.package-name.url = "github:xFA25E/package-name";
  outputs = { self, package-name, nixpkgs }: let
    pkgs = import nixpkgs {
      system = "your-system";
      overlays = [ package-name.overlays.default ];
    };
  in {
    # YOUR OUTPUTS
  };
}
#+end_src
